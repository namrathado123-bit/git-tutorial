import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { createPageUrl } from '@/utils';
import { base44 } from '@/api/base44Client';
import { Search, Plus, Minus, ShoppingCart, ChevronRight } from 'lucide-react';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import BottomNav from '@/components/layout/BottomNav';
import { cn } from '@/lib/utils';
import { toast } from 'sonner';

export default function Billing() {
  const [products, setProducts] = useState([]);
  const [filteredProducts, setFilteredProducts] = useState([]);
  const [cart, setCart] = useState([]);
  const [search, setSearch] = useState('');
  const [isDark, setIsDark] = useState(false);
  const [loading, setLoading] = useState(true);
  const [checkingOut, setCheckingOut] = useState(false);
  const navigate = useNavigate();

  useEffect(() => {
    loadData();
  }, []);

  useEffect(() => {
    const savedTheme = localStorage.getItem('smartbiz_theme');
    if (savedTheme) setIsDark(savedTheme === 'dark');
  }, []);

  useEffect(() => {
    if (search) {
      setFilteredProducts(
        products.filter(p => 
          p.name?.toLowerCase().includes(search.toLowerCase()) ||
          p.slNo?.toLowerCase().includes(search.toLowerCase()) ||
          p.category?.toLowerCase().includes(search.toLowerCase())
        )
      );
    } else {
      setFilteredProducts(products);
    }
  }, [search, products]);

  const loadData = async () => {
    try {
      const productsData = await base44.entities.Product.list('-created_date', 500);
      setProducts(productsData || []);
      setFilteredProducts(productsData || []);
    } catch (error) {
      toast.error('Failed to load products');
    } finally {
      setLoading(false);
    }
  };

  const getCartItem = (productId) => {
    return cart.find(item => item.productId === productId);
  };

  const addToCart = (product) => {
    const existingItem = getCartItem(product.id);
    
    if ((product.stock || 0) < 1) {
      toast.error('Out of stock');
      return;
    }

    if (existingItem) {
      if (existingItem.quantity >= (product.stock || 0)) {
        toast.error('Not enough stock');
        return;
      }
      setCart(cart.map(item => 
        item.productId === product.id 
          ? { ...item, quantity: item.quantity + 1 }
          : item
      ));
    } else {
      setCart([...cart, {
        productId: product.id,
        productName: product.name,
        price: product.sellPrice || 0,
        quantity: 1,
        maxStock: product.stock || 0
      }]);
    }
  };

  const removeFromCart = (productId) => {
    const existingItem = getCartItem(productId);
    if (existingItem.quantity === 1) {
      setCart(cart.filter(item => item.productId !== productId));
    } else {
      setCart(cart.map(item => 
        item.productId === productId 
          ? { ...item, quantity: item.quantity - 1 }
          : item
      ));
    }
  };

  const getTotalItems = () => {
    return cart.reduce((sum, item) => sum + item.quantity, 0);
  };

  const getTotalAmount = () => {
    return cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
  };

  const handleCheckout = async () => {
    if (cart.length === 0) {
      toast.error('Cart is empty');
      return;
    }

    setCheckingOut(true);
    try {
      const billNumber = INV${Date.now()};
      const items = cart.map(item => ({
        productId: item.productId,
        productName: item.productName,
        quantity: item.quantity,
        price: item.price,
        total: item.price * item.quantity
      }));

      const subtotal = getTotalAmount();
      const tax = subtotal * 0.05; // 5% tax
      const total = subtotal + tax;

      const bill = await base44.entities.Bill.create({
        billNumber,
        items,
        subtotal,
        tax,
        total,
        status: 'Completed'
      });

      // Update inventory - delete if stock becomes 0 or negative
      for (const item of cart) {
        const product = products.find(p => p.id === item.productId);
        if (product) {
          const newStock = (product.stock || 0) - item.quantity;
          if (newStock <= 0) {
            // Auto-delete product when stock reaches 0
            await base44.entities.Product.delete(product.id);
          } else {
            await base44.entities.Product.update(product.id, {
              stock: newStock
            });
          }
        }
      }

      setCart([]);
      await loadData(); // Reload products to reflect stock changes/deletions
      toast.success('Bill created successfully!');
      navigate(createPageUrl('InvoiceDetail') + ?id=${bill.id});
    } catch (error) {
      toast.error('Failed to create bill');
      console.error(error);
    } finally {
      setCheckingOut(false);
    }
  };

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="w-12 h-12 border-4 border-indigo-600 border-t-transparent rounded-full animate-spin" />
      </div>
    );
  }

  return (
    <div className={cn(
      "min-h-screen pb-32 transition-colors",
      isDark ? "bg-slate-900" : "bg-gradient-to-br from-slate-50 to-indigo-50"
    )}>
      {/* Header */}
      <div className={cn(
        "sticky top-0 z-30 backdrop-blur-xl border-b shadow-lg",
        isDark ? "bg-slate-900/95 border-slate-800/50" : "bg-white/95 border-gray-200/50"
      )}>
        <div className="max-w-lg mx-auto px-6 py-5">
          <h1 className={cn("text-2xl font-bold mb-1", isDark ? "text-white" : "text-gray-900")}>
            Point of Sale
          </h1>
          <p className={cn("text-sm font-medium", isDark ? "text-gray-400" : "text-gray-600")}>
            Select products to create a bill
          </p>
        </div>
      </div>

      <div className="max-w-lg mx-auto px-6 py-6 space-y-6">
        {/* Search with icon inside */}
        <div className="relative">
          <Search className={cn(
            "absolute left-4 top-1/2 -translate-y-1/2 w-5 h-5",
            isDark ? "text-[#4F9BFF]" : "text-indigo-500"
          )} />
          <Input
            placeholder="Search products..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            className={cn(
              "pl-12 h-14 rounded-2xl shadow-lg transition-all duration-300 focus:shadow-xl font-medium",
              isDark ? "bg-slate-800/80 backdrop-blur-xl border-slate-700/50 text-white placeholder:text-gray-500" : "bg-white/90 backdrop-blur-sm border-gray-200/50 placeholder:text-gray-400"
            )}
          />
        </div>

        {/* Products Grid with better design */}
        <div className="space-y-4">
          {filteredProducts.map((product) => {
            const cartItem = getCartItem(product.id);
            const inStock = (product.stock || 0) > 0;

            return (
              <div
                key={product.id}
                className={cn(
                  "rounded-3xl p-6 transition-all duration-300 hover:shadow-2xl hover:-translate-y-2 active:scale-98 group",
                  isDark 
                    ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50 shadow-[0px_8px_20px_rgba(0,0,0,0.3)] hover:shadow-[0_0_25px_rgba(139,92,246,0.3)]" 
                    : "bg-white/90 backdrop-blur-sm border border-gray-200/50 shadow-[0px_4px_12px_rgba(0,0,0,0.08)] hover:shadow-[0px_12px_28px_rgba(0,0,0,0.12)]"
                )}
              >
                <div className="flex items-center justify-between">
                  <div className="flex-1 min-w-0 flex items-center gap-4">
                    <div className={cn(
                      "p-3 rounded-2xl flex-shrink-0 shadow-lg transition-all duration-300 group-hover:scale-110 group-hover:rotate-6",
                      isDark ? "bg-gradient-to-br from-purple-500/30 to-pink-500/30" : "bg-gradient-to-br from-purple-100 to-pink-100"
                    )}>
                      <ShoppingCart className={cn("w-6 h-6", isDark ? "text-[#A06BFF]" : "text-purple-600")} />
                    </div>
                    <div className="flex-1 min-w-0">
                      <h3 className={cn("font-semibold text-base truncate mb-2", isDark ? "text-white" : "text-gray-900")}>
                        {product.name}
                      </h3>
                      <div className="flex items-center gap-3 flex-wrap">
                        <span className={cn("text-2xl font-bold tracking-tight", isDark ? "text-[#27D5C3]" : "text-emerald-600")}>
                          ₹{product.sellPrice || 0}
                        </span>
                        <span className={cn(
                          "text-xs px-3 py-1.5 rounded-xl font-bold shadow-md",
                          inStock
                            ? isDark ? "bg-emerald-500/20 text-[#27D5C3] border border-emerald-500/30" : "bg-emerald-100 text-emerald-700 border border-emerald-200"
                            : isDark ? "bg-red-500/20 text-[#FF4DA6] border border-red-500/30" : "bg-red-100 text-red-700 border border-red-200"
                        )}>
                          Stock: {product.stock || 0}
                        </span>
                      </div>
                    </div>
                  </div>

                  {/* Add/Remove Buttons */}
                  <div className="flex items-center gap-2 ml-4">
                    {cartItem ? (
                      <>
                        <Button
                          size="icon"
                          variant="outline"
                          onClick={() => removeFromCart(product.id)}
                          className={cn("h-10 w-10 rounded-xl shadow-md hover:shadow-lg transition-all duration-300 hover:scale-110 active:scale-95", isDark && "border-slate-600")}
                        >
                          <Minus className="w-4 h-4" />
                        </Button>
                        <span className={cn("font-bold text-lg w-10 text-center", isDark ? "text-white" : "text-gray-900")}>
                          {cartItem.quantity}
                        </span>
                        <Button
                          size="icon"
                          variant="outline"
                          onClick={() => addToCart(product)}
                          className={cn("h-10 w-10 rounded-xl shadow-md hover:shadow-lg transition-all duration-300 hover:scale-110 active:scale-95", isDark && "border-slate-600")}
                          disabled={cartItem.quantity >= (product.stock || 0)}
                        >
                          <Plus className="w-4 h-4" />
                        </Button>
                      </>
                    ) : (
                      <Button
                        onClick={() => addToCart(product)}
                        disabled={!inStock}
                        className="rounded-2xl bg-gradient-to-r from-indigo-600 to-purple-600 hover:from-indigo-700 hover:to-purple-700 shadow-lg hover:shadow-xl transition-all duration-300 hover:scale-110 active:scale-95 font-semibold px-6"
                      >
                        <Plus className="w-4 h-4 mr-1" />
                        Add
                      </Button>
                    )}
                  </div>
                </div>
              </div>
            );
          })}

          {filteredProducts.length === 0 && (
            <div className="text-center py-12">
              <p className={cn("text-lg", isDark ? "text-gray-400" : "text-gray-600")}>
                No products found
              </p>
            </div>
          )}
        </div>
      </div>

      {/* Floating Cart with gradient button */}
      {cart.length > 0 && (
        <div className={cn(
          "fixed bottom-20 left-0 right-0 z-40 border-t backdrop-blur-xl shadow-2xl",
          isDark ? "bg-slate-900/95 border-slate-800/50" : "bg-white/95 border-gray-200/50"
        )}>
          <div className="max-w-lg mx-auto px-6 py-5">
            <div className="flex items-center justify-between">
              <div>
                <p className={cn("text-sm font-semibold mb-1", isDark ? "text-gray-400" : "text-gray-600")}>
                  {getTotalItems()} items
                </p>
                <p className={cn("text-3xl font-bold tracking-tight", isDark ? "text-white" : "text-gray-900")}>
                  ₹{getTotalAmount().toFixed(2)}
                </p>
              </div>
              <Button
                onClick={handleCheckout}
                disabled={checkingOut}
                className="h-14 px-10 rounded-2xl bg-gradient-to-r from-emerald-600 to-teal-600 hover:from-emerald-700 hover:to-teal-700 shadow-[0px_8px_20px_rgba(16,185,129,0.4)] hover:shadow-[0px_12px_28px_rgba(16,185,129,0.5)] transition-all duration-300 hover:-translate-y-1 active:scale-95 font-bold text-base"
              >
                {checkingOut ? 'Processing...' : 'Checkout'}
                <ChevronRight className="w-5 h-5 ml-2" />
              </Button>
            </div>
          </div>
        </div>
      )}

      <BottomNav isDark={isDark} />
    </div>
  );
}
