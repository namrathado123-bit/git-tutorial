import React, { useState, useEffect } from 'react';
import { useNavigate } from 'react-router-dom';
import { createPageUrl } from '@/utils';
import { base44 } from '@/api/base44Client';
import { DollarSign, TrendingUp, Package, Users, Plus, Upload, BarChart3, AlertTriangle } from 'lucide-react';
import { Button } from '@/components/ui/button';
import BottomNav from '@/components/layout/BottomNav';
import ProfilePanel from '@/components/layout/ProfilePanel';
import KPICard from '@/components/dashboard/KPICard';
import SalesChart from '@/components/dashboard/SalesChart';
import CriticalAlert from '@/components/dashboard/CriticalAlert';
import { cn } from '@/lib/utils';

export default function Home() {
  const [user, setUser] = useState(null);
  const [products, setProducts] = useState([]);
  const [bills, setBills] = useState([]);
  const [isDark, setIsDark] = useState(false);
  const [showProfile, setShowProfile] = useState(false);
  const [loading, setLoading] = useState(true);
  const navigate = useNavigate();

  useEffect(() => {
    loadData();
  }, []);

  useEffect(() => {
    const savedTheme = localStorage.getItem('smartbiz_theme');
    if (savedTheme) {
      setIsDark(savedTheme === 'dark');
    }
  }, []);

  const loadData = async () => {
    try {
      const currentUser = await base44.auth.me();
      setUser(currentUser);
      
      if (!currentUser.hasCompletedOnboarding) {
        navigate(createPageUrl('Onboarding'));
        return;
      }

      if (currentUser.themePreference) {
        setIsDark(currentUser.themePreference === 'dark');
      }

      const [productsData, billsData] = await Promise.all([
        base44.entities.Product.list('-created_date', 100),
        base44.entities.Bill.list('-created_date', 50)
      ]);

      setProducts(productsData || []);
      setBills(billsData || []);
    } catch (error) {
      console.error('Failed to load data:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleToggleTheme = async () => {
    const newTheme = !isDark;
    setIsDark(newTheme);
    localStorage.setItem('smartbiz_theme', newTheme ? 'dark' : 'light');
    
    try {
      await base44.auth.updateMe({ themePreference: newTheme ? 'dark' : 'light' });
    } catch (error) {
      console.error('Failed to save theme:', error);
    }
  };

  // Calculate KPIs
  const last7DaysBills = bills.filter(bill => {
    const billDate = new Date(bill.created_date);
    const weekAgo = new Date();
    weekAgo.setDate(weekAgo.getDate() - 7);
    return billDate >= weekAgo;
  });

  const totalSales = last7DaysBills.reduce((sum, bill) => sum + (bill.total || 0), 0);
  const avgProfit = products.length > 0
    ? products.reduce((sum, p) => sum + ((p.sellPrice || 0) - (p.costPrice || 0)), 0) / products.length
    : 0;
  const profitMargin = avgProfit > 0 ? ((avgProfit / (avgProfit + (products[0]?.costPrice || 1))) * 100) : 0;
  const lowStockCount = products.filter(p => (p.stock || 0) < (p.lowStockThreshold || 10)).length;
  const inventoryHealth = products.length > 0
    ? ((products.length - lowStockCount) / products.length * 100)
    : 100;

  // Generate chart data
  const chartData = Array.from({ length: 7 }, (_, i) => {
    const date = new Date();
    date.setDate(date.getDate() - (6 - i));
    const dayBills = bills.filter(bill => {
      const billDate = new Date(bill.created_date);
      return billDate.toDateString() === date.toDateString();
    });
    const dayTotal = dayBills.reduce((sum, bill) => sum + (bill.total || 0), 0);
    
    return {
      day: date.toLocaleDateString('en-US', { weekday: 'short' }),
      actual: Math.round(dayTotal),
      forecast: Math.round(dayTotal * 1.1)
    };
  });

  // Critical alerts
  const alerts = [
    ...(lowStockCount > 0 ? [{
      title: ${lowStockCount} Products Low Stock,
      description: 'Immediate restocking required to avoid stockouts',
      priority: 'critical',
      type: 'stock',
      action: () => navigate(createPageUrl('Inventory'))
    }] : []),
    {
      title: 'Demand Forecast Spike',
      description: '45% chance of increased demand in next 3 days',
      priority: 'warning',
      type: 'demand',
      action: () => navigate(createPageUrl('Insights'))
    }
  ];

  if (loading) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-slate-50 to-indigo-50 flex items-center justify-center">
        <div className="text-center">
          <div className="w-12 h-12 border-4 border-indigo-600 border-t-transparent rounded-full animate-spin mx-auto mb-4" />
          <p className="text-gray-600">Loading...</p>
        </div>
      </div>
    );
  }

  return (
    <div className={cn(
      "min-h-screen pb-20 transition-colors duration-300",
      isDark ? "bg-slate-900" : "bg-gradient-to-br from-slate-50 via-white to-indigo-50"
    )}>
      {/* Header with personality */}
      <div className={cn(
        "sticky top-0 z-30 backdrop-blur-xl border-b shadow-lg",
        isDark ? "bg-slate-900/95 border-slate-800/50" : "bg-white/95 border-gray-200/50"
      )}>
        <div className="max-w-lg mx-auto px-6 py-5">
            <div className="flex items-center justify-between">
              <div>
                <h1 className={cn("text-2xl font-bold bg-gradient-to-r from-indigo-600 to-purple-600 bg-clip-text text-transparent mb-1", isDark && "from-indigo-400 to-purple-400")}>
                  Welcome back, {user?.full_name?.split(' ')[0] || 'there'} ðŸ‘‹
                </h1>
                <p className={cn("text-sm font-medium", isDark ? "text-gray-400" : "text-gray-600")}>
                  Let's grow {user?.businessName || 'your business'} today!
                </p>
              </div>
            <button
              onClick={() => setShowProfile(true)}
              className={cn(
                "w-12 h-12 rounded-full flex items-center justify-center text-xl font-bold transition-all duration-300 hover:scale-110 hover:shadow-xl active:scale-95 shadow-lg",
                isDark ? "bg-indigo-500/20 text-indigo-400 hover:shadow-indigo-500/30" : "bg-indigo-100 text-indigo-600 hover:shadow-indigo-200"
              )}
            >
              {user?.profilePictureUrl ? (
                <img src={user.profilePictureUrl} alt="Profile" className="w-full h-full rounded-full object-cover" />
              ) : (
                user?.full_name?.[0]?.toUpperCase() || 'U'
              )}
            </button>
          </div>
        </div>
      </div>

      <div className="max-w-lg mx-auto px-6 py-6 space-y-8">
        {/* Action Buttons with gradients */}
        <div className="grid grid-cols-2 gap-4">
          <Button
            onClick={() => navigate(createPageUrl('Billing'))}
            className="h-16 rounded-2xl bg-gradient-to-r from-indigo-600 to-purple-600 hover:from-indigo-700 hover:to-purple-700 shadow-[0px_8px_20px_rgba(99,102,241,0.35)] hover:shadow-[0px_16px_32px_rgba(99,102,241,0.5)] hover:-translate-y-1 transition-all duration-300 active:scale-95 font-semibold text-base"
          >
            <Plus className="w-5 h-5 mr-2" />
            New Bill
          </Button>
          <Button
            onClick={() => navigate(createPageUrl('UploadData'))}
            variant="outline"
            className={cn(
              "h-16 rounded-2xl border-2 shadow-[0px_4px_12px_rgba(0,0,0,0.08)] hover:shadow-[0px_8px_20px_rgba(0,0,0,0.12)] hover:-translate-y-1 transition-all duration-300 active:scale-95 font-semibold text-base",
              isDark ? "border-slate-700 hover:bg-slate-800 hover:border-slate-600" : "border-indigo-200 hover:bg-indigo-50 hover:border-indigo-300"
            )}
          >
            <Upload className="w-5 h-5 mr-2" />
            Upload Data
          </Button>
        </div>

        {/* Section title */}
        <div>
          <h2 className={cn("text-lg font-bold mb-4", isDark ? "text-white" : "text-gray-900")}>
            Today's Insights
          </h2>
        
        {/* KPI Cards */}
        <div className="grid grid-cols-2 gap-4">
          <KPICard
            title="Sales (7d)"
            value={â‚¹${totalSales.toLocaleString()}}
            change={12.5}
            icon={DollarSign}
            color="bg-emerald-500"
            isDark={isDark}
            onClick={() => navigate(createPageUrl('Reports'))}
          />
          <KPICard
            title="Profit Margin"
            value={${profitMargin.toFixed(1)}%}
            change={8.2}
            icon={TrendingUp}
            color="bg-blue-500"
            isDark={isDark}
            onClick={() => navigate(createPageUrl('Reports'))}
          />
          <KPICard
            title="Inventory Health"
            value={${inventoryHealth.toFixed(0)}%}
            change={inventoryHealth >= 80 ? 5 : -3}
            icon={Package}
            color={inventoryHealth >= 80 ? "bg-green-500" : "bg-amber-500"}
            isDark={isDark}
            onClick={() => navigate(createPageUrl('Inventory'))}
          />
          <KPICard
            title="New Customers"
            value={last7DaysBills.length}
            change={15.3}
            icon={Users}
            color="bg-purple-500"
            isDark={isDark}
            onClick={() => navigate(createPageUrl('Reports'))}
          />
        </div>
        </div>

        {/* Sales Chart */}
        <SalesChart data={chartData} isDark={isDark} />

        {/* Critical Highlights */}
        {alerts.length > 0 && (
          <div className="space-y-4">
            <div className="flex items-center gap-3">
              <div className={cn("p-3 rounded-2xl shadow-lg", isDark ? "bg-amber-500/20" : "bg-amber-100")}>
                <AlertTriangle className={cn("w-6 h-6", isDark ? "text-amber-400" : "text-amber-600")} />
              </div>
              <h2 className={cn("font-bold text-lg", isDark ? "text-white" : "text-gray-900")}>
                Critical Highlights
              </h2>
            </div>
            {alerts.map((alert, index) => (
              <CriticalAlert
                key={index}
                {...alert}
                isDark={isDark}
                onClick={alert.action}
              />
            ))}
          </div>
        )}

        {/* Quick Actions with vibrant icons */}
        <div>
          <h2 className={cn("font-bold text-lg mb-4", isDark ? "text-white" : "text-gray-900")}>
            Quick Actions
          </h2>
          <div className="grid grid-cols-2 gap-4">
            <button
              onClick={() => navigate(createPageUrl('Reports'))}
              className={cn(
                "p-6 rounded-3xl transition-all duration-300 hover:scale-105 hover:shadow-2xl hover:-translate-y-2 active:scale-95 text-left group",
                isDark 
                  ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50 shadow-[0px_8px_20px_rgba(0,0,0,0.3)] hover:shadow-[0_0_25px_rgba(79,155,255,0.3)]" 
                  : "bg-white/90 backdrop-blur-sm border border-gray-200/50 shadow-[0px_4px_12px_rgba(0,0,0,0.08)] hover:shadow-[0px_12px_28px_rgba(0,0,0,0.12)]"
              )}
            >
              <div className={cn("p-3 rounded-2xl inline-block mb-4 shadow-lg transition-all duration-300 group-hover:scale-110 group-hover:rotate-6", isDark ? "bg-gradient-to-br from-indigo-500/30 to-purple-500/30" : "bg-gradient-to-br from-indigo-100 to-purple-100")}>
                <BarChart3 className={cn("w-7 h-7", isDark ? "text-[#4F9BFF]" : "text-indigo-600")} />
              </div>
              <p className={cn("font-semibold text-base", isDark ? "text-white" : "text-gray-900")}>
                Reports
              </p>
              <p className={cn("text-xs mt-1 font-medium", isDark ? "text-gray-400" : "text-gray-600")}>
                View analytics
              </p>
            </button>
            <button
              onClick={() => navigate(createPageUrl('Inventory'))}
              className={cn(
                "p-6 rounded-3xl transition-all duration-300 hover:scale-105 hover:shadow-2xl hover:-translate-y-2 active:scale-95 text-left group",
                isDark 
                  ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50 shadow-[0px_8px_20px_rgba(0,0,0,0.3)] hover:shadow-[0_0_25px_rgba(39,213,195,0.3)]" 
                  : "bg-white/90 backdrop-blur-sm border border-gray-200/50 shadow-[0px_4px_12px_rgba(0,0,0,0.08)] hover:shadow-[0px_12px_28px_rgba(0,0,0,0.12)]"
              )}
            >
              <div className={cn("p-3 rounded-2xl inline-block mb-4 shadow-lg transition-all duration-300 group-hover:scale-110 group-hover:rotate-6", isDark ? "bg-gradient-to-br from-emerald-500/30 to-teal-500/30" : "bg-gradient-to-br from-emerald-100 to-teal-100")}>
                <Package className={cn("w-7 h-7", isDark ? "text-[#27D5C3]" : "text-emerald-600")} />
              </div>
              <p className={cn("font-semibold text-base", isDark ? "text-white" : "text-gray-900")}>
                Stock
              </p>
              <p className={cn("text-xs mt-1 font-medium", isDark ? "text-gray-400" : "text-gray-600")}>
                Manage inventory
              </p>
            </button>
          </div>
        </div>
      </div>

      <ProfilePanel
        isOpen={showProfile}
        onClose={() => setShowProfile(false)}
        user={user}
        onUpdate={loadData}
        isDark={isDark}
        onToggleTheme={handleToggleTheme}
      />

      <BottomNav isDark={isDark} />
    </div>
  );
}
