import React, { useState, useEffect } from 'react';
import { base44 } from '@/api/base44Client';
import { Package, AlertTriangle, TrendingDown, Search, Edit2, Trash2 } from 'lucide-react';
import { Input } from '@/components/ui/input';
import { Badge } from '@/components/ui/badge';
import { Button } from '@/components/ui/button';
import BottomNav from '@/components/layout/BottomNav';
import StockEditInput from '@/components/inventory/StockEditInput';
import DeleteConfirmModal from '@/components/inventory/DeleteConfirmModal';
import { cn } from '@/lib/utils';
import { toast } from 'sonner';

export default function Inventory() {
  const [products, setProducts] = useState([]);
  const [filteredProducts, setFilteredProducts] = useState([]);
  const [search, setSearch] = useState('');
  const [isDark, setIsDark] = useState(false);
  const [loading, setLoading] = useState(true);
  const [editingProductId, setEditingProductId] = useState(null);
  const [deleteModalOpen, setDeleteModalOpen] = useState(false);
  const [productToDelete, setProductToDelete] = useState(null);

  useEffect(() => {
    loadData();
  }, []);

  useEffect(() => {
    const savedTheme = localStorage.getItem('smartbiz_theme');
    if (savedTheme) setIsDark(savedTheme === 'dark');
  }, []);

  useEffect(() => {
    if (search) {
      setFilteredProducts(
        products.filter(p => 
          p.name?.toLowerCase().includes(search.toLowerCase()) ||
          p.slNo?.toLowerCase().includes(search.toLowerCase()) ||
          p.category?.toLowerCase().includes(search.toLowerCase())
        )
      );
    } else {
      setFilteredProducts(products);
    }
  }, [search, products]);

  const loadData = async () => {
    try {
      const productsData = await base44.entities.Product.list('-created_date', 1000);
      setProducts(productsData || []);
      setFilteredProducts(productsData || []);
    } catch (error) {
      console.error('Failed to load products:', error);
    } finally {
      setLoading(false);
    }
  };

  const calculateStats = () => {
    const total = products.length;
    const lowStock = products.filter(p => (p.stock || 0) < (p.lowStockThreshold || 10)).length;
    const outOfStock = products.filter(p => (p.stock || 0) === 0).length;
    const fulfillmentRate = total > 0 ? (((total - outOfStock) / total) * 100) : 100;

    return { total, lowStock, outOfStock, fulfillmentRate };
  };

  const getStockStatus = (product) => {
    const stock = product.stock || 0;
    const threshold = product.lowStockThreshold || 10;

    if (stock === 0) return { label: 'Out of Stock', color: 'bg-red-500' };
    if (stock < threshold) return { label: 'Low Stock', color: 'bg-amber-500' };
    return { label: 'In Stock', color: 'bg-emerald-500' };
  };

  const handleUpdateStock = async (productId, newStock) => {
    try {
      if (newStock <= 0) {
        // Auto-delete when stock is 0 or negative
        await base44.entities.Product.delete(productId);
        setProducts(prev => prev.filter(p => p.id !== productId));
        toast.success('Product removed (stock was 0)');
      } else {
        // Update stock
        await base44.entities.Product.update(productId, { stock: newStock });
        setProducts(prev => prev.map(p => 
          p.id === productId ? { ...p, stock: newStock } : p
        ));
        toast.success('Stock updated successfully');
      }
      setEditingProductId(null);
    } catch (error) {
      toast.error('Failed to update stock');
      console.error(error);
    }
  };

  const handleDeleteProduct = async (productId) => {
    try {
      await base44.entities.Product.delete(productId);
      setProducts(prev => prev.filter(p => p.id !== productId));
      toast.success('Product deleted successfully');
      setDeleteModalOpen(false);
      setProductToDelete(null);
    } catch (error) {
      toast.error('Failed to delete product');
      console.error(error);
    }
  };

  const openDeleteModal = (product) => {
    setProductToDelete(product);
    setDeleteModalOpen(true);
  };

  const stats = calculateStats();

  if (loading) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="w-12 h-12 border-4 border-indigo-600 border-t-transparent rounded-full animate-spin" />
      </div>
    );
  }

  return (
    <div className={cn(
      "min-h-screen pb-20 transition-colors",
      isDark ? "bg-slate-900" : "bg-gradient-to-br from-slate-50 to-indigo-50"
    )}>
      {/* Header */}
      <div className={cn(
        "sticky top-0 z-30 backdrop-blur-xl border-b shadow-lg",
        isDark ? "bg-slate-900/95 border-slate-800/50" : "bg-white/95 border-gray-200/50"
      )}>
        <div className="max-w-lg mx-auto px-6 py-5">
          <h1 className={cn("text-2xl font-bold mb-1", isDark ? "text-white" : "text-gray-900")}>
            Inventory
          </h1>
          <p className={cn("text-sm font-medium", isDark ? "text-gray-400" : "text-gray-600")}>
            Manage your stock levels
          </p>
        </div>
      </div>

      <div className="max-w-lg mx-auto px-6 py-6 space-y-6">
        {/* Section title */}
        <h2 className={cn("text-lg font-bold", isDark ? "text-white" : "text-gray-900")}>
          Stock Overview
        </h2>
        
        {/* Stats with improved design */}
        <div className="grid grid-cols-3 gap-4">
          <div className={cn(
            "rounded-2xl p-5 text-center shadow-lg transition-all duration-300 hover:shadow-xl hover:-translate-y-1",
            isDark ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50" : "bg-white/90 backdrop-blur-sm border border-gray-200/50"
          )}>
            <p className={cn("text-3xl font-bold tracking-tight", isDark ? "text-[#27D5C3]" : "text-emerald-600")}>
              {stats.fulfillmentRate.toFixed(0)}%
            </p>
            <p className={cn("text-xs mt-2 font-semibold", isDark ? "text-gray-400" : "text-gray-600")}>
              Fulfillment
            </p>
          </div>
          <div className={cn(
            "rounded-2xl p-5 text-center shadow-lg transition-all duration-300 hover:shadow-xl hover:-translate-y-1",
            isDark ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50" : "bg-white/90 backdrop-blur-sm border border-gray-200/50"
          )}>
            <p className={cn("text-3xl font-bold tracking-tight", isDark ? "text-[#F7FF3D]" : "text-amber-600")}>
              {stats.lowStock}
            </p>
            <p className={cn("text-xs mt-2 font-semibold", isDark ? "text-gray-400" : "text-gray-600")}>
              Low Stock
            </p>
          </div>
          <div className={cn(
            "rounded-2xl p-5 text-center shadow-lg transition-all duration-300 hover:shadow-xl hover:-translate-y-1",
            isDark ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50" : "bg-white/90 backdrop-blur-sm border border-gray-200/50"
          )}>
            <p className={cn("text-3xl font-bold tracking-tight", isDark ? "text-[#FF4DA6]" : "text-red-600")}>
              {stats.outOfStock}
            </p>
            <p className={cn("text-xs mt-2 font-semibold", isDark ? "text-gray-400" : "text-gray-600")}>
              Out of Stock
            </p>
          </div>
        </div>

        {/* Search with icon inside */}
        <div>
          <h2 className={cn("text-lg font-bold mb-4", isDark ? "text-white" : "text-gray-900")}>
            Your Inventory
          </h2>
          <div className="relative">
            <Search className={cn(
              "absolute left-4 top-1/2 -translate-y-1/2 w-5 h-5",
              isDark ? "text-[#4F9BFF]" : "text-indigo-500"
            )} />
            <Input
              placeholder="Search products..."
              value={search}
              onChange={(e) => setSearch(e.target.value)}
              className={cn(
                "pl-12 h-14 rounded-2xl shadow-lg transition-all duration-300 focus:shadow-xl font-medium",
                isDark ? "bg-slate-800/80 backdrop-blur-xl border-slate-700/50 text-white placeholder:text-gray-500" : "bg-white/90 backdrop-blur-sm border-gray-200/50 placeholder:text-gray-400"
              )}
            />
          </div>
        </div>

        {/* Products List with better spacing */}
        <div className="space-y-3">
          {filteredProducts.map((product) => {
            const status = getStockStatus(product);
            
            return (
              <div
                key={product.id}
                className={cn(
                  "rounded-3xl p-5 transition-all duration-300 hover:shadow-2xl hover:-translate-y-1 active:scale-98",
                  isDark 
                    ? "bg-slate-800/80 backdrop-blur-xl border border-slate-700/50 shadow-[0px_6px_16px_rgba(0,0,0,0.3)] hover:shadow-[0_0_20px_rgba(79,155,255,0.2)]" 
                    : "bg-white/90 backdrop-blur-sm border border-gray-200/50 shadow-[0px_4px_12px_rgba(0,0,0,0.08)] hover:shadow-[0px_12px_24px_rgba(0,0,0,0.12)]"
                )}
              >
                <div className="flex items-start justify-between mb-3">
                  <div className="flex-1 min-w-0">
                    <h3 className={cn("font-semibold truncate", isDark ? "text-white" : "text-gray-900")}>
                      {product.name}
                    </h3>
                    {product.category && (
                      <p className={cn("text-sm", isDark ? "text-gray-400" : "text-gray-600")}>
                        {product.category}
                      </p>
                    )}
                  </div>
                  <div className="flex items-center gap-2 ml-2">
                    <Badge className={cn(
                      "rounded-xl px-3 py-1 font-semibold text-xs shadow-md",
                      status.color === 'bg-red-500' && (isDark ? "bg-red-500/20 text-[#FF4DA6] border-red-500/30" : "bg-red-100 text-red-700 border-red-200"),
                      status.color === 'bg-amber-500' && (isDark ? "bg-amber-500/20 text-[#F7FF3D] border-amber-500/30" : "bg-amber-100 text-amber-700 border-amber-200"),
                      status.color === 'bg-emerald-500' && (isDark ? "bg-emerald-500/20 text-[#27D5C3] border-emerald-500/30" : "bg-emerald-100 text-emerald-700 border-emerald-200")
                    )}>
                      {status.label}
                    </Badge>
                    <Button
                      size="icon"
                      variant="ghost"
                      onClick={() => openDeleteModal(product)}
                      className={cn(
                        "h-8 w-8 rounded-xl hover:bg-red-500/20 transition-all duration-300 hover:scale-110",
                        isDark ? "text-red-400" : "text-red-600"
                      )}
                    >
                      <Trash2 className="w-4 h-4" />
                    </Button>
                  </div>
                </div>

                <div className="grid grid-cols-3 gap-4 text-sm mt-3">
                  <div>
                    <div className="flex items-center justify-between mb-1">
                      <p className={cn("text-xs font-medium", isDark ? "text-gray-500" : "text-gray-400")}>
                        Stock
                      </p>
                      {editingProductId !== product.id && (
                        <Button
                          size="icon"
                          variant="ghost"
                          onClick={() => setEditingProductId(product.id)}
                          className={cn(
                            "h-6 w-6 rounded-lg hover:bg-indigo-500/20 transition-all duration-300",
                            isDark ? "text-indigo-400" : "text-indigo-600"
                          )}
                        >
                          <Edit2 className="w-3 h-3" />
                        </Button>
                      )}
                    </div>
                    {editingProductId === product.id ? (
                      <StockEditInput
                        initialStock={product.stock || 0}
                        onSave={(newStock) => handleUpdateStock(product.id, newStock)}
                        onCancel={() => setEditingProductId(null)}
                        isDark={isDark}
                      />
                    ) : (
                      <p className={cn("font-bold text-base", isDark ? "text-white" : "text-gray-900")}>
                        {product.stock || 0}
                      </p>
                    )}
                  </div>
                  <div>
                    <p className={cn("text-xs font-medium mb-1", isDark ? "text-gray-500" : "text-gray-400")}>
                      Sell Price
                    </p>
                    <p className={cn("font-bold text-base", isDark ? "text-[#27D5C3]" : "text-emerald-600")}>
                      ₹{product.sellPrice || 0}
                    </p>
                  </div>
                  <div>
                    <p className={cn("text-xs font-medium mb-1", isDark ? "text-gray-500" : "text-gray-400")}>
                      MRP
                    </p>
                    <p className={cn("font-bold text-base", isDark ? "text-white" : "text-gray-900")}>
                      ₹{product.mrp || 0}
                    </p>
                  </div>
                </div>

                {product.slNo && (
                  <p className={cn("text-xs mt-2", isDark ? "text-gray-500" : "text-gray-400")}>
                    SKU: {product.slNo}
                  </p>
                )}
              </div>
            );
          })}

          {filteredProducts.length === 0 && (
            <div className="text-center py-12">
              <Package className={cn("w-16 h-16 mx-auto mb-4", isDark ? "text-gray-600" : "text-gray-400")} />
              <p className={cn("text-lg", isDark ? "text-gray-400" : "text-gray-600")}>
                No products found
              </p>
            </div>
          )}
        </div>
      </div>

      <DeleteConfirmModal
        isOpen={deleteModalOpen}
        onClose={() => {
          setDeleteModalOpen(false);
          setProductToDelete(null);
        }}
        onConfirm={() => handleDeleteProduct(productToDelete.id)}
        productName={productToDelete?.name || ''}
        isDark={isDark}
      />

      <BottomNav isDark={isDark} />
    </div>
  );
}
